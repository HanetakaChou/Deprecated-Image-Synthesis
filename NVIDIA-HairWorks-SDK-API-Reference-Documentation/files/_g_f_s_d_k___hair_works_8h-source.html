00029 /*
00030 
00031 -----------------------------------------------------------------------------
00032                                    HairWorks                                 
00033 -----------------------------------------------------------------------------
00034 
00035 NVIDIA HairWorks enables users to simulate and render fur/hair to provide a 
00036 truly interactive game experience. The technology originates from multiple 
00037 simulation and rendering technologies that NVIDIA has developed in the past. 
00038 This combines rendering technologies for human hair (GDC 2008) as well as 
00039 various simulation technologies shown in multiple events (Fur Demo - GDC2012).
00040 
00041 To make the hair/fur technologies a product, NVIDIA engineers have put
00042 together these existing technologies and added new techniques to ensure a 
00043 smooth pipeline for a variety of characters.
00044 
00045 -----------------------------------------------------------------------------
00046                              INTEGRATION EXAMPLE                             
00047 -----------------------------------------------------------------------------
00048 
00049 This is an overview of the integration. Please refer to the API documentation
00050 in the header file and the sample code for details.
00051 
00052 (1.) LOAD THE LIBRARY
00053 
00054     #include "GFSDK_HairWorks.h" // hairworks main header file
00055 
00056     GFSDK_HairSDK* g_pHairSDK = GFSDK_LoadHairSDK("GFSDK_HairWorks.win64.dll",
00057                                                   GFSDK_HAIRWORKS_VERSION);
00058 
00059 (2.) INITIALIZE
00060 
00061 Initialize D3D Device.
00062 
00063     g_pHairSDK->InitRenderResources(pd3dDevice);
00064 
00065 Create the hair asset manually from your asset data
00066 
00067     GFSDK_HairAssetDescriptor hairAssetDescriptor;
00068     hairAssetDescriptor.m_NumGuideHairs = 4;
00069     hairAssetDescriptor.m_NumVertices = 20;
00070     hairAssetDescriptor.m_pVertices = vertices;
00071     hairAssetDescriptor.m_pEndIndices = endIndices;
00072     hairAssetDescriptor.m_pBindPoses = bindPoses;
00073     hairAssetDescriptor.m_pBoneParents = boneParents;
00074     ...
00075     g_pHairSDK->CreateHairAsset(g_hairAssetDescriptor, &g_hairAssetID);
00076 
00077 Or load it from the .apx file (generated by DCC tool or FurViewer) using the API.
00078     
00079     g_pHairSDK->LoadHairAssetFromFile("test.apx", &g_hairAssetID);
00080 
00081 Create a hair instance from the hair asset.
00082 
00083     g_pHairSDK->CreateHairInstance(g_hairAssetID, &hairInstanceID);
00084 
00085 (3.) SIMULATION/RENDERING
00086 
00087 Set render context for HairWorks.
00088 
00089     g_pHairSDK->SetCurrentContext(pd3dContext);
00090 
00091 Update descriptor parameters for each hair instance.
00092 
00093     GFSDK_HairInstanceDescriptor hairInstanceDescriptor;
00094     hairInstanceDescriptor.m_width = 0.2;
00095     hairInstanceDescriptor.m_density = 1.0;
00096     hairInstanceDescriptor.m_lengthNoise = 0.0f;
00097     hairInstanceDescriptor.m_simulate = true;
00098     ...
00099     g_pHairSDK->UpdateInstanceDescriptor(hairInstanceID, hairInstanceDescriptor);
00100 
00101 Update skinning matrices for each hair
00102     g_pHairSDK->UpdateSkinningMatrices(hairInstanceID, numBones, pMatrices); 
00103 
00104 Run simulation for all hairs.
00105 
00106     g_pHairSDK->StepSimulation(fElapsedTime);
00107 
00108 Set view matrix and projection matrix.
00109 
00110     D3DXMATRIX projection = *g_Camera.GetProjMatrix();
00111     D3DXMATRIX view = *g_Camera.GetViewMatrix();
00112     g_pHairSDK->SetViewProjection((const gfsdk_float4x4*)&view,(const gfsdk_float4x4*)&projection);
00113 
00114 Set your hair pixel shader before rendering hairs.
00115 
00116     pd3dContext->PSSetShader(g_customHairWorksShader, NULL, 0);
00117 
00118 Render each hair instance.
00119 
00120     g_pHairSDK->RenderHairs(hairInstanceID);
00121 
00122 (4.) RELEASE
00123 
00124     g_pHairSDK->FreeHairInstance(hairInstanceID);
00125     g_pHairSDK->FreeHairAsset(hairAssetID);
00126     g_pHairSDK->FreeRenderResources();
00127     g_pHairSDK->Release();
00128 
00129 */
